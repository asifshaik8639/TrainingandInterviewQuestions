function nonRepeatingChar(string)
{
  for(var i=0;i<string.length;i++){
    var ch = string.charAt(i);
    if(string.indexOf(ch) == i && string.indexOf(ch,i+1) == -1){
        return ch;
       }
  }
  return null;
}
